build fftw3 on linux:
cmake .. -DBUILD_STATIC_LIBS=ON -DENABLE_FLOAT=ON -D CMAKE_C_FLAGS="-fPIC"

build fftw3 on windows:
cmake .. -G "Visual Studio 16 2019" -A x64 -DBUILD_STATIC_LIBS=ON -DBUILD_SHARED_LIBS=OFF -DENABLE_FLOAT=ON
 
build fftw on mac:
cmake .. -D CMAKE_OSX_ARCHITECTURES="arm64;x86_64" -D XCODE_ATTRIBUTE_ENABLE_HARDENED_RUNTIME=YES -DBUILD_STATIC_LIBS=ON -DBUILD_SHARED_LIBS=OFF -DENABLE_FLOAT=ON -DCMAKE_OSX_DEPLOYMENT_TARGET=10.13
